* Request Lifecycle (Sequence Diagram)

#+begin_src plantuml :file request-sequence.png
@startuml
title Request Execution Flow

actor Client

participant Router
participant Handler
participant RiskSignals
participant Authorizer
participant UseCase
participant Repository
database DB

Client -> Router : HTTP Request
Router -> Handler : route()

Handler -> RiskSignals : build()
RiskSignals --> Handler : context

Handler -> Authorizer : authorize(resource, action, context)
Authorizer --> Handler : decision OK

Handler -> UseCase : execute()

UseCase -> Repository : load/save
Repository -> DB : query
DB --> Repository
Repository --> UseCase

UseCase --> Handler : result
Handler --> Client : HTTP Response

@enduml
